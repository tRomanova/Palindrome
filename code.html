The code

                        
                            //entering function
                            // This function is responsible for receiving and responding to the user's string
                            function getValues() {
                                let inputString = document.getElementById('strValue').value;
                            
                                let checkedString = checkForPalindrome(inputString);
                                displayResults(checkedString);
                            }
                            
                            //logic function
                            // This function is responsible determining whether a string is a palindrome
                            function checkForPalindrome(strToCheck, reversedString) {
                                let reg = /[^A-Za-z0-9]/g;
                                let newString = ""
                                let palindrome = true;
                            
                                let regString = strToCheck.toLowerCase().replace(reg, "");
                            
                                let halfString = regString.length / 2;
                                let i = 0;
                            
                                for (let j = regString.length - 1; j >= halfString; j--) {
                            
                                    if (regString[i] === regString[j]) {
                                        i += 1;
                                    } else {
                                        newString = "isNot"
                                        palindrome = false;
                                    }
                                }
                                if (palindrome) {
                                    newString = regString
                                }
                            
                                return newString;
                            }
                            
                            //This function is responsible for displaying a message to the user to show 
                            //whether their string is a palindrome or not
                            function displayResults(checkedString) {
                                let className;
                            
                                if (checkedString !== "isNot") {
                                    className = 'alertPalindrome';
                                    document.getElementById('results').innerHTML = `
Well Done! You entered a Palindrome!

                                        
Your phrase reversed is:  ${checkedString} 
`;
                                    document.getElementById('alert').classList.remove('invisible');
                                } else {
                                    className = 'alertNotPalindrom';
                                    document.getElementById('results').innerHTML = `
Sorry! You entered not a Palindrome!

                                    
Try again!
`;
                                    document.getElementById('alert').classList.remove('invisible');
                                }
                            }

The code is structured in 3 functions

  • getValues: This function grabs the user input. Call function 2 other functions that checks if the user input is a palindrome or not, and displays the result.
  • checkForPalindrome: This function uses regular expression to ignore non alphanumeric character, converts string into lower case, and iterates the string in a for loop to check if it is a palindrome or not.
  • displayResults: This function displays an alert in green or red based on whether or not it is a palindrome.
